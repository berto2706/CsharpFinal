namespace CsharpEx03;

class Program
{
    static void Main(string[] args)
    {
        ApresentarPrograma();
        Calculo();
    }

    public static void ApresentarPrograma()
    {
        Console.WriteLine("Programa: Números do início ao fim");
    }

    public static void Calculo()
    {
        Console.Write("Início: ");
        int inicio = Convert.ToInt32(Console.ReadLine());

        Console.Write("Fim: ");
        int fim = Convert.ToInt32(Console.ReadLine());

        for (int i = inicio; i <= fim; i++)
            Console.WriteLine(i);
    }
}
